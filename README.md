from langchain_community.vectorstores import Chroma
from langchain_community.embeddings import OpenAIEmbeddings

openai.api_key = "YOUR_API_KEY"

# -----------------------------
# Load documents
# -----------------------------
def load_documents():
    docs = []

    # Load PDFs
    pdf_loader = PyPDFLoader("data/sample.pdf")
    docs.extend(pdf_loader.load())

    # Load text files
    text_loader = TextLoader("data/notes.txt")
    docs.extend(text_loader.load())

    return docs

# -----------------------------
# Create / load vector database
# -----------------------------
def create_vector_store(docs):
    embeddings = OpenAIEmbeddings()
    vectordb = Chroma.from_documents(docs, embedding=embeddings, persist_directory="chroma_db")
    vectordb.persist()
    return vectordb

# -----------------------------
# RAG Query Function
# -----------------------------
def rag_query(question):
    vectordb = Chroma(persist_directory="chroma_db",
                      embedding_function=OpenAIEmbeddings())

    results = vectordb.similarity_search(question, k=3)

    context = "\n\n".join([doc.page_content for doc in results])

    response = openai.chat.completions.create(
        model="gpt-4.1-mini",
        messages=[
            {"role": "system", "content": "You are a helpful AI that uses context to answer questions."},
            {"role": "user", "content": f"Context:\n{context}\n\nQuestion: {question}"}
        ]
    )

    return response.choices[0].message["content"]

# -----------------------------
# Main App
# -----------------------------
if __name__ == "__main__":
    print("Loading documents...")
    docs = load_documents()

    print("Creating vector store...")
    create_vector_store(docs)

    print("Ready! Ask a question:")
    while True:
        user_input = input("> ")
        answer = rag_query(user_input)
        print("\nAI:", answer, "\n")
